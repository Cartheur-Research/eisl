# Stat
Stat library

# Usage
(import "stat")

# Spec

### (mean ls)
e.g. (mean '(1 2 3)) => 2

### (median ls)
e.g. (median '(1 2 3)) => 2

### (mode ls)
e.g. (mode '(2 3 1 2)) => 2

### (variance ls)
e.g. (variance '(1 2 3)) => 0.666667

### (sd ls)
standard deviation
e.g. (sd '(1 2 3)) => 0.816497
